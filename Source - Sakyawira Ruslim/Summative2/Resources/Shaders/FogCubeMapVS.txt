layout (location = 0) in vec3 position; 

out vec3 fragTexCoord;
out vec4 mWorldPos;

void main()
{
	// gl_Position =  MVP * vec4(position, 1.0);

	mWorldPos = model *vec4(position, 1.0); 
	gl_Position =  (MVP*(inverse(model))) * mWorldPos;

	fragTexCoord = position;
}
